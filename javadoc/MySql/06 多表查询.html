06 еӨҡиЎЁжҹҘиҜў вҖӢ

еӨҡиЎЁжҹҘиҜўпјҢд№ҹз§°дёәе…іиҒ”жҹҘиҜўпјҢжҢҮдёӨдёӘжҲ–жӣҙеӨҡдёӘиЎЁдёҖиө·е®ҢжҲҗжҹҘиҜўж“ҚдҪңгҖӮ

еүҚжҸҗжқЎд»¶пјҡиҝҷдәӣдёҖиө·жҹҘиҜўзҡ„иЎЁд№Ӣй—ҙжҳҜжңүе…ізі»зҡ„пјҲдёҖеҜ№дёҖгҖҒдёҖеҜ№еӨҡпјүпјҢе®ғд»¬д№Ӣй—ҙдёҖе®ҡжңүе…іиҒ”еӯ—ж®өгҖӮжҜ”еҰӮпјҡе‘ҳе·ҘиЎЁе’ҢйғЁй—ЁиЎЁпјҢиҝҷдёӨдёӘиЎЁдҫқйқ вҖңйғЁй—Ёзј–еҸ·вҖқиҝӣ иЎҢе…іиҒ”гҖӮ

дёҖдёӘжЎҲдҫӢеј•еҸ‘зҡ„еӨҡиЎЁиҝһжҺҘ вҖӢ

жЎҲдҫӢиҜҙжҳҺ вҖӢ

shell
select count(department_id) from departments; # 27

еҒҮи®ҫ вҖӢ

еҒҮи®ҫжҲ‘жғіжҹҘиҜўе‘ҳе·Ҙзҡ„е§“еҗҚеҸҠе…¶йғЁй—ЁеҗҚз§°пјҢз„¶еҗҺжҲ‘еҶҷзҡ„ sql еғҸдёӢйқўиҝҷж ·пјҡ

shell
SELECT last_name, department_name
FROM employees, departments;

жҹҘиҜўз»“жһңеҸ‘зҺ°жңү 2889 жқЎж•°жҚ®гҖӮиҖҢ 107Г—27 еҲҡеҘҪзӯүдәҺ 2889гҖӮ

еҲҶжһҗй”ҷиҜҜжғ…еҶөпјҡ

shell
SELECT COUNT(employee_id) FROM employees;
# иҫ“еҮә 107 иЎҢ
SELECT COUNT(department_id)FROM departments;
# иҫ“еҮә 27 иЎҢ
SELECT 107*27 FROM dual

жҲ‘д»¬жҠҠдёҠиҝ°еӨҡиЎЁжҹҘиҜўдёӯеҮәзҺ°зҡ„й—®йўҳз§°дёәвҖңз¬ӣеҚЎе°”з§Ҝзҡ„й”ҷиҜҜвҖқгҖӮ

з¬ӣеҚЎе°”з§ҜпјҲжҲ–дәӨеҸүиҝһжҺҘпјүзҡ„зҗҶи§Ј вҖӢ

з¬ӣеҚЎе°”д№ҳз§ҜжҳҜдёҖдёӘж•°еӯҰиҝҗз®—гҖӮеҒҮи®ҫжҲ‘жңүдёӨдёӘйӣҶеҗҲ X е’Ң YпјҢйӮЈд№Ҳ X е’Ң Y зҡ„з¬ӣеҚЎе°”з§Ҝе°ұжҳҜ X е’Ң Y зҡ„жүҖжңүеҸҜиғҪ з»„еҗҲпјҢд№ҹе°ұжҳҜз¬¬дёҖдёӘеҜ№иұЎжқҘиҮӘдәҺ XпјҢз¬¬дәҢдёӘеҜ№иұЎжқҘиҮӘдәҺ Y зҡ„жүҖжңүеҸҜиғҪгҖӮз»„еҗҲзҡ„дёӘж•°еҚідёәдёӨдёӘйӣҶеҗҲдёӯе…ғзҙ  дёӘж•°зҡ„д№ҳз§Ҝж•°гҖӮ

SQL92 и§„иҢғдёӯпјҢз¬ӣеҚЎе°”з§Ҝд№ҹз§°дёәдәӨеҸүиҝһжҺҘпјҢиӢұж–ҮжҳҜ CROSS JOINгҖӮеңЁ SQL99 дёӯд№ҹжҳҜдҪҝз”Ё CROSS JOIN иЎЁзӨәдәӨ еҸүиҝһжҺҘгҖӮе®ғзҡ„дҪңз”Ёе°ұжҳҜеҸҜд»ҘжҠҠд»»ж„ҸиЎЁиҝӣиЎҢиҝһжҺҘпјҢеҚідҪҝиҝҷдёӨеј иЎЁдёҚзӣёе…ігҖӮеңЁ MySQL дёӯеҰӮдёӢжғ…еҶөдјҡеҮәзҺ°з¬ӣеҚЎ е°”з§Ҝпјҡ

shell
# жҹҘиҜўе‘ҳе·Ҙе§“еҗҚе’ҢжүҖеңЁйғЁй—ЁеҗҚз§°
SELECT last_name,department_name FROM employees,departments;
SELECT last_name,department_name FROM employees CROSS JOIN departments;
SELECT last_name,department_name FROM employees INNER JOIN departments;
SELECT last_name,department_name FROM employees JOIN departments;

жЎҲдҫӢеҲҶжһҗдёҺй—®йўҳи§ЈеҶі вҖӢ

з¬ӣеҚЎе°”з§Ҝзҡ„й”ҷиҜҜдјҡеңЁдёӢйқўжқЎд»¶дёӢдә§з”ҹпјҡ

  • зңҒз•ҘеӨҡдёӘиЎЁзҡ„иҝһжҺҘжқЎд»¶пјҲжҲ–е…іиҒ”жқЎд»¶пјү
  • иҝһжҺҘжқЎд»¶пјҲжҲ–е…іиҒ”жқЎд»¶пјүж— ж•Ҳ
  • жүҖжңүиЎЁдёӯзҡ„жүҖжңүиЎҢдә’зӣёиҝһжҺҘ

дёәдәҶйҒҝе…Қз¬ӣеҚЎе°”з§ҜпјҢеҸҜд»ҘеңЁ WHERE еҠ е…Ҙжңүж•Ҳзҡ„иҝһжҺҘжқЎд»¶гҖӮ

еҠ е…ҘиҝһжҺҘжқЎд»¶зҡ„жҹҘиҜўиҜӯжі•пјҡ

shell
SELECT  table1.column, table2.column
FROM    table1, table2
WHERE   table1.column1 = table2.column2;  # еңЁ WHERE еӯҗ еҸҘдёӯеҶҷе…ҘиҝһжҺҘжқЎд»¶

и§ЈеҶіеҠһжі•пјҡ

shell
# жЎҲдҫӢпјҡжҹҘиҜўе‘ҳе·Ҙзҡ„е§“еҗҚеҸҠе…¶йғЁй—ЁеҗҚз§°
SELECT last_name, department_name
FROM employees, departments
WHERE employees.department_id = departments.department_id;

еңЁиЎЁдёӯжңүзӣёеҗҢеҲ—ж—¶пјҢиҰҒеңЁеҲ—еҗҚд№ӢеүҚеҠ дёҠиЎЁеҗҚеүҚзјҖгҖӮ

еӨҡиЎЁжҹҘиҜўеҲҶзұ»и®Іи§Ј вҖӢ

еҲҶзұ» 1пјҢзӯүеҖјиҝһжҺҘ vs йқһзӯүеҖјиҝһжҺҘ вҖӢ

зӯүеҖјиҝһжҺҘ вҖӢ

where жқЎд»¶еҝ…йЎ»жҳҜдёӨдёӘиЎЁзҡ„еӯ—ж®өзӣёеҗҢгҖӮ

shell
SELECT employees.employee_id, employees.last_name, 
       employees.department_id, departments.department_id,
       departments.location_id
FROM   employees, departments
WHERE  employees.department_id = departments.department_id;

еӨҡдёӘиЎЁдёӯжңүзӣёеҗҢеҲ—ж—¶пјҢеҝ…йЎ»еңЁеҲ—еҗҚд№ӢеүҚеҠ дёҠиЎЁеҗҚеүҚзјҖгҖӮеңЁдёҚеҗҢиЎЁдёӯе…·жңүзӣёеҗҢеҲ—еҗҚзҡ„еҲ—еҸҜд»Ҙз”ЁиЎЁеҗҚеҠ д»ҘеҢәеҲҶгҖӮ

shell
SELECT employees.last_name, departments.department_name,employees.department_id
FROM employees, departments
WHERE employees.department_id = departments.department_id;

дҪҝз”ЁеҲ«еҗҚеҸҜд»Ҙз®ҖеҢ–жҹҘиҜўгҖӮеҲ—еҗҚеүҚдҪҝз”ЁиЎЁеҗҚеүҚзјҖеҸҜд»ҘжҸҗй«ҳжҹҘиҜўж•ҲзҺҮгҖӮ йңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢеҰӮжһңжҲ‘д»¬дҪҝз”ЁдәҶиЎЁзҡ„еҲ«еҗҚпјҢеңЁжҹҘиҜўеӯ—ж®өдёӯгҖҒиҝҮж»ӨжқЎд»¶дёӯе°ұеҸӘиғҪдҪҝз”ЁеҲ«еҗҚиҝӣиЎҢд»ЈжӣҝпјҢ дёҚиғҪдҪҝз”ЁеҺҹжңүзҡ„иЎЁеҗҚпјҢеҗҰеҲҷе°ұдјҡжҠҘй”ҷгҖӮ

shell
SELECT e.employee_id, e.last_name, e.department_id,
       d.department_id, d.location_id
FROM   employees e , departments d
WHERE  e.department_id = d.department_id;

иҝһжҺҘ n дёӘиЎЁпјҢиҮіе°‘йңҖиҰҒ n-1 дёӘиҝһжҺҘжқЎд»¶гҖӮжҜ”еҰӮпјҢиҝһжҺҘдёүдёӘиЎЁпјҢиҮіе°‘йңҖиҰҒдёӨдёӘиҝһжҺҘжқЎд»¶гҖӮ

йқһзӯүеҖјиҝһжҺҘ вҖӢ

shell
SELECT e.last_name, e.salary, j.grade_level
FROM   employees e, job_grades j
WHERE  e.salary BETWEEN j.lowest_sal AND j.highest_sal;

еҲҶзұ» 2пјҢиҮӘиҝһжҺҘ vs йқһиҮӘиҝһжҺҘ вҖӢ

еҪ“ table1 е’Ң table2 жң¬иҙЁдёҠжҳҜеҗҢдёҖеј иЎЁпјҢеҸӘжҳҜз”ЁеҸ–еҲ«еҗҚзҡ„ж–№ејҸиҷҡжӢҹжҲҗдёӨеј иЎЁд»Ҙд»ЈиЎЁдёҚеҗҢзҡ„ж„Ҹд№үгҖӮз„¶еҗҺдёӨ дёӘиЎЁеҶҚиҝӣиЎҢеҶ…иҝһжҺҘпјҢеӨ–иҝһжҺҘзӯүжҹҘиҜўгҖӮ

shell
SELECT CONCAT(worker.last_name ,' works for ' 
       , manager.last_name)
FROM   employees worker, employees manager
WHERE  worker.manager_id = manager.employee_id ;

еҲҶзұ» 3пјҢеҶ…иҝһжҺҘ vs еӨ–иҝһжҺҘ вҖӢ

йҷӨдәҶжҹҘиҜўж»Ўи¶іжқЎд»¶зҡ„и®°еҪ•д»ҘеӨ–пјҢеӨ–иҝһжҺҘиҝҳеҸҜд»ҘжҹҘиҜўжҹҗдёҖж–№дёҚж»Ўи¶іжқЎд»¶зҡ„и®°еҪ•гҖӮ

  • еҶ…иҝһжҺҘпјҡз»“жһңйӣҶдёӯеҸӘеҢ…еҗ«дёӨеј иЎЁеҢ№й…Қзҡ„иЎҢпјҢдёҚеҢ…еҗ«дёҚеҢ№й…Қзҡ„иЎҢ
  • еӨ–иҝһжҺҘпјҡз»“жһңйӣҶдёӯйҷӨдәҶеҢ…еҗ«дёӨеј иЎЁеҢ№й…Қзҡ„иЎҢеӨ–пјҢиҝҳеҢ…еҗ«дёҚеҢ№й…Қзҡ„иЎҢпјҢиҝҷз§ҚиҝһжҺҘз§°дёәе·ҰпјҲжҲ–еҸіпјүеӨ–иҝһжҺҘгҖӮжІЎжңүеҢ№й…Қзҡ„иЎҢз»“жһңиЎЁдёӯзӣёеә”зҡ„еҲ—дёәз©ә (NULL)
  • еҰӮжһңжҳҜе·ҰеӨ–иҝһжҺҘпјҢеҲҷиҝһжҺҘжқЎд»¶дёӯе·Ұиҫ№зҡ„иЎЁд№ҹз§°дёәдё»иЎЁпјҢеҸіиҫ№зҡ„иЎЁз§°дёәд»ҺиЎЁгҖӮ еҰӮжһңжҳҜеҸіеӨ–иҝһжҺҘпјҢеҲҷиҝһжҺҘжқЎд»¶дёӯеҸіиҫ№зҡ„иЎЁд№ҹз§°дёәдё»иЎЁпјҢе·Ұиҫ№зҡ„иЎЁз§°дёәд»ҺиЎЁ

еңЁ SQL92 дёӯйҮҮз”ЁпјҲ+пјүд»ЈиЎЁд»ҺиЎЁжүҖеңЁзҡ„дҪҚзҪ®гҖӮеҚіе·ҰжҲ–еҸіеӨ–иҝһжҺҘдёӯпјҢ(+) иЎЁзӨәе“ӘдёӘжҳҜд»ҺиЎЁгҖӮ

Oracle еҜ№ SQL92 ж”ҜжҢҒиҫғеҘҪпјҢиҖҢ MySQL еҲҷдёҚж”ҜжҢҒ SQL92 зҡ„еӨ–иҝһжҺҘгҖӮ

shell
#е·ҰеӨ–иҝһжҺҘ
SELECT last_name,department_name
FROM employees ,departments
WHERE employees.department_id = departments.department_id(+);
#еҸіеӨ–иҝһжҺҘ
SELECT last_name,department_name
FROM employees ,departments
WHERE employees.department_id(+) = departments.department_id;

иҖҢдё”еңЁ SQL92 дёӯпјҢеҸӘжңүе·ҰеӨ–иҝһжҺҘе’ҢеҸіеӨ–иҝһжҺҘпјҢжІЎжңүж»ЎпјҲжҲ–е…ЁпјүеӨ–иҝһжҺҘгҖӮ

MySql дёӯ SQL99 иҜӯжі•е®һзҺ°еӨҡиЎЁжҹҘиҜў вҖӢ

еҹәжң¬иҜӯжі• вҖӢ

дҪҝз”Ё JOIN...ON еӯҗеҸҘеҲӣе»әиҝһжҺҘзҡ„иҜӯжі•з»“жһ„пјҡ

shell
SELECT table1.column, table2.column,table3.column
FROM table1
    JOIN table2 ON table1 е’Ң table2 зҡ„иҝһжҺҘжқЎд»¶
        JOIN table3 ON table2 е’Ң table3 зҡ„иҝһжҺҘжқЎд»¶

е®ғзҡ„еөҢеҘ—йҖ»иҫ‘зұ»дјјжҲ‘д»¬дҪҝз”Ёзҡ„ FOR еҫӘзҺҜпјҡ

shell
for t1 in table1:
    for t2 in table2:
       if condition1:
           for t3 in table3:
              if condition2:
                  output t1 + t2 + t3

SQL99 йҮҮз”Ёзҡ„иҝҷз§ҚеөҢеҘ—з»“жһ„йқһеёёжё…зҲҪгҖҒеұӮж¬ЎжҖ§жӣҙејәгҖҒеҸҜиҜ»жҖ§жӣҙејәпјҢеҚідҪҝеҶҚеӨҡзҡ„иЎЁиҝӣиЎҢиҝһжҺҘд№ҹйғҪжё…жҷ° еҸҜи§ҒгҖӮеҰӮжһңдҪ йҮҮз”Ё SQL92пјҢеҸҜиҜ»жҖ§е°ұдјҡеӨ§жү“жҠҳжүЈгҖӮ

иҜӯжі•иҜҙжҳҺпјҡ

  • еҸҜд»ҘдҪҝз”Ё ON еӯҗеҸҘжҢҮе®ҡйўқеӨ–зҡ„иҝһжҺҘжқЎд»¶
  • иҝҷдёӘиҝһжҺҘжқЎд»¶жҳҜдёҺе…¶е®ғжқЎд»¶еҲҶејҖзҡ„
  • ON еӯҗеҸҘдҪҝиҜӯеҸҘе…·жңүжӣҙй«ҳзҡ„жҳ“иҜ»жҖ§
  • е…ій”®еӯ— JOINгҖҒINNER JOINгҖҒCROSS JOIN зҡ„еҗ«д№үжҳҜдёҖж ·зҡ„пјҢйғҪиЎЁзӨәеҶ…иҝһжҺҘ

еҶ…иҝһжҺҘ (INNER JOIN) зҡ„е®һзҺ° вҖӢ

иҜӯжі•пјҡ

shell
SELECT еӯ—ж®өеҲ—иЎЁ
FROM AиЎЁ INNER JOIN BиЎЁ
ON е…іиҒ”жқЎд»¶
WHERE зӯүе…¶д»–еӯҗеҸҘ;

shell

йҷ„еҪ•пјҡеёёз”Ёзҡ„ SQL ж ҮеҮҶжңүе“Әдәӣ вҖӢ

еңЁжӯЈејҸејҖе§Ӣи®ІиҝһжҺҘиЎЁзҡ„з§Қзұ»ж—¶пјҢжҲ‘д»¬йҰ–е…ҲйңҖиҰҒзҹҘйҒ“ SQL еӯҳеңЁдёҚеҗҢзүҲжң¬зҡ„ж ҮеҮҶи§„иҢғпјҢеӣ дёәдёҚеҗҢи§„иҢғдёӢзҡ„иЎЁ иҝһжҺҘж“ҚдҪңжҳҜжңүеҢәеҲ«зҡ„гҖӮ

SQL жңүдёӨдёӘдё»иҰҒзҡ„ж ҮеҮҶпјҢеҲҶеҲ«жҳҜ SQL92 е’Ң SQL99гҖӮ92 е’Ң 99 д»ЈиЎЁдәҶж ҮеҮҶжҸҗеҮәзҡ„ж—¶й—ҙпјҢSQL92 е°ұжҳҜ 92 е№ҙ жҸҗеҮәзҡ„ж ҮеҮҶи§„иҢғгҖӮеҪ“з„¶йҷӨдәҶ SQL92 е’Ң SQL99 д»ҘеӨ–пјҢиҝҳеӯҳеңЁ SQL-86гҖҒSQL-89гҖҒSQL:2003гҖҒSQL:2008гҖҒ SQL:2011 е’Ң SQL:2016 зӯүе…¶д»–зҡ„ж ҮеҮҶгҖӮ

иҝҷд№ҲеӨҡж ҮеҮҶпјҢеҲ°еә•иҜҘеӯҰд№ е“ӘдёӘе‘ўпјҹе®һйҷ…дёҠжңҖйҮҚиҰҒзҡ„ SQL ж ҮеҮҶе°ұжҳҜ SQL92 е’Ң SQL99гҖӮдёҖиҲ¬жқҘиҜҙ SQL92 зҡ„ еҪўејҸжӣҙз®ҖеҚ•пјҢдҪҶжҳҜеҶҷзҡ„ SQL иҜӯеҸҘдјҡжҜ”иҫғй•ҝпјҢеҸҜиҜ»жҖ§иҫғе·®гҖӮиҖҢ SQL99 зӣёжҜ”дәҺ SQL92 жқҘиҜҙпјҢиҜӯжі•жӣҙеҠ еӨҚжқӮпјҢ дҪҶеҸҜиҜ»жҖ§жӣҙејәгҖӮжҲ‘д»¬д»ҺиҝҷдёӨдёӘж ҮеҮҶеҸ‘еёғзҡ„йЎөж•°д№ҹиғҪзңӢеҮәпјҢSQL92 зҡ„ж ҮеҮҶжңү 500 йЎөпјҢиҖҢ SQL99 ж ҮеҮҶи¶…иҝҮдәҶ 1000 йЎөгҖӮе®һйҷ…дёҠд»Һ SQL99 д№ӢеҗҺпјҢеҫҲе°‘жңүдәәиғҪжҺҢжҸЎжүҖжңүеҶ…е®№пјҢеӣ дёәзЎ®е®һеӨӘеӨҡдәҶгҖӮе°ұеҘҪжҜ”жҲ‘д»¬дҪҝз”Ё WindowsгҖҒLinux е’Ң Office зҡ„ж—¶еҖҷпјҢеҫҲе°‘жңүдәәиғҪжҺҢжҸЎе…ЁйғЁеҶ…е®№дёҖж ·гҖӮжҲ‘д»¬еҸӘйңҖиҰҒжҺҢжҸЎдёҖдәӣж ёеҝғзҡ„еҠҹиғҪпјҢж»Ў и¶іж—Ҙеёёе·ҘдҪңзҡ„йңҖжұӮеҚіеҸҜгҖӮ

SQL92 е’Ң SQL99 жҳҜз»Ҹе…ёзҡ„ SQL ж ҮеҮҶпјҢд№ҹеҲҶеҲ«еҸ«еҒҡ SQL-2 е’Ң SQL-3 ж ҮеҮҶгҖӮд№ҹжӯЈжҳҜеңЁиҝҷдёӨдёӘж ҮеҮҶеҸ‘еёғд№Ӣ еҗҺпјҢSQL еҪұе“ҚеҠӣи¶ҠжқҘи¶ҠеӨ§пјҢз”ҡиҮіи¶…и¶ҠдәҶж•°жҚ®еә“йўҶеҹҹгҖӮзҺ°еҰӮд»Ҡ SQL е·Із»ҸдёҚд»…д»…жҳҜж•°жҚ®еә“йўҶеҹҹзҡ„дё»жөҒиҜӯиЁҖпјҢ иҝҳжҳҜдҝЎжҒҜйўҶеҹҹдёӯдҝЎжҒҜеӨ„зҗҶзҡ„дё»жөҒиҜӯиЁҖгҖӮеңЁеӣҫеҪўжЈҖзҙўгҖҒеӣҫеғҸжЈҖзҙўд»ҘеҸҠиҜӯйҹіжЈҖзҙўдёӯйғҪиғҪзңӢеҲ° SQL иҜӯиЁҖзҡ„дҪҝ з”ЁгҖӮ

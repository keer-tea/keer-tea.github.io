const messages = {
  en: {
    i18n_tpje7hyr_1651741146224_info: 'жөӢиҜ•%{var}'
    // ... дјҡжңүеҫҲеӨҡзҝ»иҜ‘
  },
  zh: {
    i18n_tpje7hyr_1651741146224_info : 'жөӢиҜ•%{var}'
    // ... дјҡжңүеҫҲеӨҡзҝ»иҜ‘
  },
  ja: { 
    i18n_tpje7hyr_1651741146224_info: 'и©ҰйЁ“%{var}'
    // ... дјҡжңүеҫҲеӨҡзҝ»иҜ‘
  }
}

// йҖҡиҝҮйҖүйЎ№еҲӣе»ә VueI18n е®һдҫӢ
const i18n = new VueI18n({
  locale: 'zh', // и®ҫзҪ®ең°еҢәпјҢй»ҳи®Өдёӯж–ҮпјҢеҜ№еә” messages дёӯзҡ„ key
  messages, // и®ҫзҪ®ең°еҢәдҝЎжҒҜ
})


// йҖҡиҝҮ `i18n` йҖүйЎ№еҲӣе»ә Vue е®һдҫӢ
new Vue({ i18n }).$mount('#app')

е…·еҗҚж јејҸ

<p>{{ $t('i18n_tpje7hyr_1651741146224_info'.replace('%var', '<span color="red">nice</span>')}}</p>
